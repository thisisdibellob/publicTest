gcc 코드파일.c -o 만들파일
./만들파일

git init
git add 파일이름   //파일 추가하겠다
git commit -m '수정기록'   //파일 저장

git add .   //모든 파일 추가
git status   //상태창 보기
git log --all --oneline   //commit한 내역 보기
git log --oneline --all --graph   //그래프로 보기

git branch 이름 
git switch 이름   //이름 브랜치로 이동


//합치게 될 메인 브랜치로 가서
git merge 합칠브랜치이름 

//merge 충돌 시 원하는 코드코 수정 후 git add / git commit 하기

//merge 완료된 브랜치는 삭제
git branch -d 브랜치이름

//merge 안한 브랜치 삭제
git branch -D 브랜치명

//새 브랜치가 중심 브랜치가 됨
git rebase 중심브랜치명   //새브랜치로가서 해야됨

//새브랜치가 메인 브랜치 합쳐짐, 근데 새브랜치 로그가 안 보이도록
git merge --squash 새브랜치이름

//깃허브 연결
git remote add origin 링크

//깃허브 연결 확인용
git remote -v

//깃허브 연결 삭제
git remote rm origin

//푸쉬
git push origin main(여긴 브랜치 이름)

//풀
git pull origin main

//강제 병합 ㄱㄱ
 git pull origin main --allow-unrelated-histories
